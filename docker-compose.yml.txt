services:
  mlflow:
    image: ghcr.io/mlflow/mlflow:latest
    command: mlflow server --backend-store-uri sqlite:///mlflow.db --default-artifact-root ./mlruns --host 0.0.0.0 --port 5000
    ports: ["5000:5000"]
    volumes:
      - ./mlruns:/mlruns
      - ./mlflow.db:/mlflow.db
      - ./:/work
    working_dir: /work

  service:
    build: .
    environment:
      - MLFLOW_TRACKING_URI=http://mlflow:5000
    ports: ["8000:8000"]
    depends_on:
      - mlflow
